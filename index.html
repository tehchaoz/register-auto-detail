Setup Guide

How to host this app:

1. Upload this HTML file to any static host (GitHub Pages, Netlify, Vercel, or even open locally).

2. That's it — all data saves in your browser's local storage.


Connect to Google Sheets:

1. Open your Google Sheet → Extensions → Apps Script.

2. Delete all existing code and paste the code block below.

3. Click Deploy → New Deployment.

4. Set Type: Web App, Execute as: Me, Who has access: Anyone → Deploy.

5. Copy the new Web App URL and paste it above.

All data is shared — entries from every user sync to the same sheet. The dashboard reads from the sheet so everyone sees the same data.

IMPORTANT: You must create a New Deployment after updating the script code.

function doPost(e) {
  try {
    var raw = e.parameter.data || (e.postData ? e.postData.contents : '');
    var d = JSON.parse(raw);

    // Handle PIN sync
    if (d.action === 'setPin' && d.ownerPin) {
      PropertiesService.getScriptProperties().setProperty('ownerPin', d.ownerPin);
      return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);
    }

    var entries = d.entries || d;
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    if (!Array.isArray(entries)) entries = [entries];
    if (entries.length === 0) {
      return ContentService.createTextOutput(JSON.stringify({status:'ok', count:0})).setMimeType(ContentService.MimeType.JSON);
    }
    var headers = ['Date','Dealership','VIN','Brand','Model','Year','Stock#','RO#','Department','Service','Price','Invoice','Notes','ID','Synced At'];

    var groups = {};
    entries.forEach(function(r) {
      var key = r.dealership || 'Unknown';
      if (!groups[key]) groups[key] = [];
      groups[key].push(r);
    });

    Object.keys(groups).forEach(function(key) {
      var groupEntries = groups[key];
      var sheet = ss.getSheetByName(key);
      if (!sheet) {
        sheet = ss.insertSheet(key);
        var title = ['Register Auto Detail — ' + key];
        while (title.length < headers.length) title.push('');
        sheet.getRange(1, 1, 1, headers.length).setValues([title]);
        sheet.getRange(1, 1, 1, headers.length).merge().setFontWeight('bold').setFontSize(14);
        sheet.getRange(2, 1, 1, headers.length).setValues([headers]).setFontWeight('bold');
      }
      var lastRow = sheet.getLastRow();
      if (lastRow < 2) {
        var titleRow = ['Register Auto Detail — ' + key];
        while (titleRow.length < headers.length) titleRow.push('');
        sheet.getRange(1, 1, 1, headers.length).setValues([titleRow]);
        sheet.getRange(1, 1, 1, headers.length).merge().setFontWeight('bold').setFontSize(14);
        sheet.getRange(2, 1, 1, headers.length).setValues([headers]).setFontWeight('bold');
        lastRow = 2;
      }
      var rows = groupEntries.map(function(r) {
        return [r.date, r.dealership, r.vin, r.brand, r.model, r.year, r.stock, r.ro, r.department, r.service, r.price, r.invoice || '', r.notes, r.id || '', new Date().toISOString()];
      });
      sheet.getRange(lastRow + 1, 1, rows.length, headers.length).setValues(rows);
    });

    return ContentService.createTextOutput(JSON.stringify({status:'ok', count: entries.length})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheets = ss.getSheets();
    var allEntries = [];
    var headers = ['Date','Dealership','VIN','Brand','Model','Year','Stock#','RO#','Department','Service','Price','Invoice','Notes','ID','Synced At'];

    // Read shared owner PIN from script properties
    var ownerPin = PropertiesService.getScriptProperties().getProperty('ownerPin') || '';

    sheets.forEach(function(sheet) {
      var name = sheet.getName();
      if (name === 'Sheet1') return;
      var data = sheet.getDataRange().getValues();
      if (data.length < 3) return;

      for (var i = 2; i < data.length; i++) {
        var row = data[i];
        allEntries.push({
          id: row[13] || '',
          date: row[0],
          dealership: row[1],
          vin: row[2],
          brand: row[3],
          model: row[4],
          year: row[5],
          stock: row[6],
          ro: row[7],
          department: row[8],
          service: row[9],
          price: row[10],
          invoice: row[11],
          notes: row[12],
        });
      }
    });

    return ContentService.createTextOutput(JSON.stringify({status:'ok', entries: allEntries, ownerPin: ownerPin})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
